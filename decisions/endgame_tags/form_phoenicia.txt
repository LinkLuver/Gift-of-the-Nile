			}
			add_country_modifier = {
				name = masters_of_the_med
				duration = -1
			}
			if = {
				limit = {
					is_iron_man = yes
					is_ai = no
					NOT = {
						has_variable = ach_tyrian_purple_var
					}
				}
				set_variable = {
					name = ach_tyrian_purple_var
				}
			}
			add_5_free_province_investments = yes
		}
		
		ai_will_do = {
			base = 1
		}
