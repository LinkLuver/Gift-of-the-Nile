namespace = me_damascus

me_damascus.1 = {
    type = country_event

    title = me_damascus_1_title
    desc = me_damascus_1_desc
    picture = great_library

    immediate = {
        random_country = {
            limit = {
                tag = PAL
            }
            save_scope_as = tadmor_scope
        }
        random_country = {
            limit = {
                tag = AGU
            }
            save_scope_as = agushi_scope
        }
        add_treasury = -100
    }

    option = {
        name = "me_damascus_1.a"
        custom_tooltip = send_invite
        trigger_event = {
            id = me_damascus.105
            days = 30
        }
        scope:tadmor_scope = {
            trigger_event = {
                id = me_damascus.102
                days = 15
            }
        }
        scope:agushi_scope = {
            trigger_event = {
                id = me_damascus.102
                days = 15
            }
        }
    }
}
me_damascus.102 = { # Invite from Damascus
    type = country_event

    title = me_damascus_102_title
    desc = me_damascus_102_desc
    picture = great_library

    immediate = {
        add_treasury = 50
    }
    option = {
        name = "me_damascus_102.b"
        trigger = {
            is_ai = yes
            ROOT = {
                any_allied_country = {
                    tag = HAM
                }
            }
        }
        custom_tooltip = join_pan_aram_congress_no
        exclusive = yes
    }
    option = {
        name = "me_damascus_102.a"
        custom_tooltip = join_pan_aram_congress_yes
        if = {
            limit = {
                tag = PAL
            }
            add_country_modifier = {
                name = tadmor_joined
                duration = 7300
            }
            c:PRY = {
                trigger_event = {
                    id = me_damascus.103
                    days = { 1 5 }
                }
            }
        }
        if = {
            limit = {
                tag = AGU
            }
            add_country_modifier = {
                name = agushi_joined
                duration = 7300
            }
            c:PRY = {
                trigger_event = {
                    id = me_damascus.104
                    days = { 1 5 }
                }
            }
        }
        ai_chance = {
            base = 100
        }
    }
    option = {
        name = "me_damascus_102.b"
        custom_tooltip = join_pan_aram_congress_no
        ai_chance = {
            base = 0
        }
    }
}

            }

        }

            name = syrion_joined
            duration = 7300
        }
    }
}
