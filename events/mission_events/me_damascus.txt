me_damascus.102 = { # Invite from Damascus
    type = country_event

    title = me_damascus_102_title
    desc = me_damascus_102_desc
    picture = great_library

    immediate = {
        add_treasury = 50
    }
    option = {
        name = "me_damascus_102.b"
        trigger = {
            is_ai = yes
            ROOT = {
                any_allied_country = {
                    tag = HAM
                }
            }
        }
        custom_tooltip = join_pan_aram_congress_no
        exclusive = yes
    }
    option = {
        name = "me_damascus_102.a"
        custom_tooltip = join_pan_aram_congress_yes
        if = {
            limit = {
                tag = PAL
            }
            add_country_modifier = {
                name = tadmor_joined
                duration = 7300
            }
            c:PRY = {
                trigger_event = {
                    id = me_damascus.103
                    days = { 1 5 }
                }
            }
        }
        if = {
            limit = {
                tag = AGU
            }
            add_country_modifier = {
                name = agushi_joined
                duration = 7300
            }
            c:PRY = {
                trigger_event = {
                    id = me_damascus.104
                    days = { 1 5 }
                }
            }
        }
        ai_chance = {
            base = 100
        }
    }
    option = {
        name = "me_damascus_102.b"
        custom_tooltip = join_pan_aram_congress_no
        ai_chance = {
            base = 0
        }
    }
}
