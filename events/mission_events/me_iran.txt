me_iran.1 = {
    type = country_event

    title = me_iran_1_title
    desc = me_iran_1_desc
    picture = great_library

    immediate = {
        every_subject = {
            limit = {
                overlord = c:PES
            }
            set_variable = annexees_persia
        }
    }

    option = {
        name = "me_iran_1.a"
        custom_tooltip = me_iran_1_tooltip
        hidden_effect = {
            every_country = {
                limit = {
                    has_variable = annexees_persia
                }
                every_owned_province = {
                    set_owned_by = c:PES
                }
            }
        }
        custom_tooltip = annex_subjects_tt
        add_treasury = 100
    }
}
