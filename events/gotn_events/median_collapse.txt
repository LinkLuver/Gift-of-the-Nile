namespace = median_collapse

median_collapse.1 = { # Initial
    type = country_event
    hidden = yes
    fire_only_once = yes
    trigger = {
        any_country = {
            tag = MAP
            OR = {
                war_with = c:PES
                region:armenia_region = {
                    any_region_province = {
                        state_loyalty <= 50
                    }
                }
                AND = {
                    current_date > 465.10.10
                    war = yes
                }
                c:ARM = {
                    has_subject_loyalty <= 35
                }
            }
        }
    }

    immediate = {
        random_country = {
            limit = {
                tag = ARM
            }
            save_scope_as = armenia_scope
        }
        random_country = {
            limit = {
                tag = MAP
            }
            #set_global_variable = median_collapse_var
        }
        if = {
            limit = {
                tag = MAP
            }
            trigger_event = {
                id = median_collapse.2
                days = 1
            }
        }
        if = {
            limit = {
                OR = {
                    tag = ARM
                    tag = AR2
                    tag = SPH
                }
            }
            trigger_event = {
                id = median_collapse.3
                days = 25
            }
        }
        if = {
            limit = {
                NOR = {
                    tag = MAP
                    tag = ARM
                    tag = AR2
                    tag = SPH
                }
            }
            trigger_event = {
                id = median_collapse.6
                days = 35
            }
        }
        if = {
            limit = {
                tag = CAP
            }
            trigger_event = {
                id = median_collapse.4
                days = 180
            }
        }
    }
}
